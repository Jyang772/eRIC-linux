10 // ---------------------------- reverse --------------------------------
11 
12 // Reverses (reflects) bits in a 32-bit word.
13 unsigned reverse(unsigned x) {
14  x = ((x & 0x55555555) << 1) | ((x >> 1) & 0x55555555);
15  x = ((x & 0x33333333) << 2) | ((x >> 2) & 0x33333333);
16  x = ((x & 0x0F0F0F0F) << 4) | ((x >> 4) & 0x0F0F0F0F);
17  x = (x << 24) | ((x & 0xFF00) << 8) |
18  ((x >> 8) & 0xFF00) | (x >> 24);
19  return x;
20 }

24 /* This is the basic CRC algorithm with no optimizations. It follows the
25 logic circuit as closely as possible. */
26 
27 unsigned int crc32a(unsigned char *message) {
28  int i, j;
29  unsigned int byte, crc;
30 
31  i = 0;
32  crc = 0xFFFFFFFF;
33  while (message[i] != 0) {
34  byte = message[i]; // Get next byte.
35  byte = reverse(byte); // 32-bit reversal.
36  for (j = 0; j <= 7; j++) { // Do eight times.
37  if ((int)(crc ^ byte) < 0)
38  crc = (crc << 1) ^ 0x04C11DB7;
39  else crc = crc << 1;
40  byte = byte << 1; // Ready next msg bit.
41  }
42  i = i + 1;
43  }
44  return reverse(~crc);
45 }

47 // ----------------------------- crc32b --------------------------------
48 
49 /* This is the basic CRC-32 calculation with some optimization but no
50 table lookup. The the byte reversal is avoided by shifting the crc reg
51 right instead of left and by using a reversed 32-bit word to represent
52 the polynomial.
53  When compiled to Cyclops with GCC, this function executes in 8 + 72n
54 instructions, where n is the number of bytes in the input message. It
55 should be doable in 4 + 61n instructions.
56  If the inner loop is strung out (approx. 5*8 = 40 instructions),
57 it would take about 6 + 46n instructions. */
58 
59 unsigned int crc32b(unsigned char *message) {
60  int i, j;
61  unsigned int byte, crc, mask;
62 
63  i = 0;
64  crc = 0xFFFFFFFF;
65  while (message[i] != 0) {
66  byte = message[i]; // Get next byte.
67  crc = crc ^ byte;
68  for (j = 7; j >= 0; j--) { // Do eight times.
69  mask = -(crc & 1);
70  crc = (crc >> 1) ^ (0xEDB88320 & mask);
71  }
72  i = i + 1;
73  }
74  return ~crc;
75 }

77 // ----------------------------- crc32c --------------------------------
78 
79 /* This is derived from crc32b but does table lookup. First the table
80 itself is calculated, if it has not yet been set up.
81 Not counting the table setup (which would probably be a separate
82 function), when compiled to Cyclops with GCC, this function executes in
83 7 + 13n instructions, where n is the number of bytes in the input
84 message. It should be doable in 4 + 9n instructions. In any case, two
85 of the 13 or 9 instrucions are load byte.
86  This is Figure 14-7 in the text. */
87 
88 unsigned int crc32c(unsigned char *message) {
89  int i, j;
90  unsigned int byte, crc, mask;
91  static unsigned int table[256];
92 
93  /* Set up the table, if necessary. */
94 
95  if (table[1] == 0) {
96  for (byte = 0; byte <= 255; byte++) {
97  crc = byte;
98  for (j = 7; j >= 0; j--) { // Do eight times.
99  mask = -(crc & 1);
100  crc = (crc >> 1) ^ (0xEDB88320 & mask);
101  }
102  table[byte] = crc;
103  }
104  }
105 
106  /* Through with table setup, now calculate the CRC. */
107 
108  i = 0;
109  crc = 0xFFFFFFFF;
110  while ((byte = message[i]) != 0) {
111  crc = (crc >> 8) ^ table[(crc ^ byte) & 0xFF];
112  i = i + 1;
113  }
114  return ~crc;
115 }

117 // ---------------------------- crc32cx --------------------------------

119 /* This is crc32b modified to load the message a fullword at a time.
120 It assumes the message is word aligned and consists of an integral
121 number of words before the 0-byte that marks the end of the message.
122  This works only on a little-endian machine.
123  Not counting the table setup (which would probably be a separate
124 function), this function should be doable in 3 + 22w instructions, where
125 w is the number of fullwords in the input message. This is equivalent to
126 3 + 5.5n instructions, where n is the number of bytes. 1.25 of those 5.5
127 instructions are loads.
128  This is Exercise 1 in the text. C.f. Christopher Dannemiller,
129 who got it from Linux Source base,
130 www.gelato.unsw.edu.au/lxr/source/lib/crc32.c, lines 105-111. */
131 
132 unsigned int crc32cx(unsigned char *message) {
133  int j;
134  unsigned int byte, crc, mask, word;
135  static unsigned int table[256];
136 
137  /* Set up the table, if necessary. */
138 
139  if (table[1] == 0) {
140  for (byte = 0; byte <= 255; byte++) {
141  crc = byte;
142  for (j = 7; j >= 0; j--) { // Do eight times.
143  mask = -(crc & 1);
144  crc = (crc >> 1) ^ (0xEDB88320 & mask);
145  }
146  table[byte] = crc;
147  }
148  }
149 
150  /* Through with table setup, now calculate the CRC. */
151 
152  crc = 0xFFFFFFFF;
153  while (((word = *(unsigned int *)message) & 0xFF) != 0) {
154  crc = crc ^ word;
155  crc = (crc >> 8) ^ table[crc & 0xFF];
156  crc = (crc >> 8) ^ table[crc & 0xFF];
157  crc = (crc >> 8) ^ table[crc & 0xFF];
158  crc = (crc >> 8) ^ table[crc & 0xFF];
159  message = message + 4;
160  }
161  return ~crc;
162 }

230 /* This is sort of like the table lookup version (crc32c), but using
231 a 16-way switch statement instead.
232  When compiled to Cyclops with GCC, this function executes in 6 + 38n
233 instructions, where n is the number of bytes in the input message. The
234 38 instructions per byte include 3 loads and 5 branches (not good). It
235 is actually 6 branches if you count the unnecessary one that GCC
236 generates because it isn't smart enough to know that the switch argument
237 cannot exceed 15. */
238 
239 unsigned int crc32e(unsigned char *message) {
240  int i, j;
241  unsigned int byte, crc, c;
242  const unsigned int g0 = 0xEDB88320, g1 = g0 >> 1,
243  g2 = g0 >> 2, g3 = g0 >> 3;
244 
245  i = 0;
246  crc = 0xFFFFFFFF;
247  while ((byte = message[i]) != 0) { // Get next byte.
248  crc = crc ^ byte;
249  for (j = 1; j >= 0; j--) { // Do two times.
250  switch(crc & 0xF) {
251  case 0: c = 0; break;
252  case 1: c = g3; break;
253  case 2: c = g2; break;
254  case 3: c = g2 ^ g3; break;
255  case 4: c = g1; break;
256  case 5: c = g1 ^ g3; break;
257  case 6: c = g1 ^ g2; break;
258  case 7: c = g1 ^ g2 ^ g3; break;
259  case 8: c = g0; break;
260  case 9: c = g0 ^ g3; break;
261  case 10: c = g0 ^ g2; break;
262  case 11: c = g0 ^ g2 ^ g3; break;
263  case 12: c = g0 ^ g1; break;
264  case 13: c = g0 ^ g1 ^ g3; break;
265  case 14: c = g0 ^ g1 ^ g2; break;
266  case 15: c = g0 ^ g1 ^ g2 ^ g3; break;
267  }
268  crc = (crc >> 4) ^ c;
269  }
270  i = i + 1;
271  }
272  return ~crc;
273 }

277 /* This is sort of like the table lookup version (crc32c), but using
278 a 256-way switch statement instead.
279  The expressions for g1, g2, ..., g7 are determined by examining what
280 the CRC-32 algorithm does to a byte of value 1, 2, 4, 8, 16, 32, 64, and
281 128, respectively. g6 and g7 are complicated because the rightmost 1-bit
282 in g0 enters the picture.
283  We rely on the compiler to evaluate, at compile time, all the
284 expressions involving the g's. They are the table values used in
285 function crc32c above (i.e., g7 = table[1], g6 = table[2], g5 =
286 table[4], etc.).
287  This idea of using a switch statement is a dumb idea if a compiler is
288 used, because the compiler (GCC anyway) implements the switch statement
289 with a 256-word label table. Thus the program still has the load from a
290 table, and it is larger than crc32c by the three words of instructions
291 used at each case statement (two instructions to load the constant, plus
292 a branch). However, since each case statement has the same amount of
293 code (three words), the label table could be avoided if the program were
294 coded in assembly language. But it would still have poor I-cache
295 performance.
296  At any rate, when compiled to Cyclops with GCC, this function
297 executes 6 + 19n instructions, where n is the number of bytes in the
298 input message. The 19 includes 2 loads and 3 branches (per byte), not
299 counting the one GCC generates to check that the switch argument doesn't
300 exceed 255 (it can't exceed 255). */

596 /* This is derived from crc32e by constructing the constant c using
597 algebraic expressions involving the rightmost four bits of the crc
598 register, rather than using a 16-way switch statement.
599  We rely on the compiler to compute the constants g>>1, g>>2, and
600 g>>3, and load them into registers ahead of the loops. Note that crc is
601 now a SIGNED integer, so the right shifts of 31 are sign-propagating
602 shifts.
603  When compiled to Cyclops with GCC, this function executes in 14 + 46n
604 instructions, where n is the number of bytes in the input message. There
605 is only one load per byte executed (but three branches). */
606 
607 unsigned int crc32g(unsigned char *message) {
608  int i, j, crc;
609  unsigned int byte, c;
610  const unsigned int g0 = 0xEDB88320, g1 = g0 >> 1,
611  g2 = g0 >> 2, g3 = g0 >> 3;
612 
613  i = 0;
614  crc = 0xFFFFFFFF;
615  while (message[i] != 0) {
616  byte = message[i]; // Get next byte.
617  crc = crc ^ byte;
618  for (j = 1; j >= 0; j--) { // Do two times.
619  c = ((crc<<31>>31) & g3) ^ ((crc<<30>>31) & g2) ^
620  ((crc<<29>>31) & g1) ^ ((crc<<28>>31) & g0);
621  crc = ((unsigned)crc >> 4) ^ c;
622  }
623  i = i + 1;
624  }
625  return ~crc;
626 }
